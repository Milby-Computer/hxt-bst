<p>/**</p>
<p></p>
<p>Get information about a user's account.</p>
<p></p>
<p>Syntax: dropboxGetAccount &lt;pAccessToken&gt;, &lt;pUserAccountID&gt;, [&lt;pCallback&gt;]</p>
<p></p>
<p>Description:</p>
<p>If the callback parameter is not empty the request will be asynchronus and</p>
<p>when complete the callback will be sent to the object that accessed the API.</p>
<p>The callback will be sent with three parameters:</p>
<p></p>
<p>- The request ID which will be the value of the it variable after calling the command</p>
<p>- The HTTP response code</p>
<p>- The data returned which will be the same data as documented for the it variable in a synchronous request.</p>
<p></p>
<p></p>
<p>Parameters:</p>
<p>pAccessToken: An OAuth2 Access token to access the user's account</p>
<p></p>
<p>pUserAccountID: The user's unique Dropbox ID</p>
<p></p>
<p>pCallback: The handler to call when the request is complete. If empty the command</p>
<p>will block until complete. Requires tsNet with LiveCode Indy License.</p>
<p></p>
<p>The result: An error if the HTTP request failed</p>
<p></p>
<p>It: For asynchronous requests it will contain a request ID. This may be used with</p>
<p>tsNet's tsNetGetStatus function. If the request is synchronous and successful the it</p>
<p>variable will be a UTF-8 encoded JSON string in the form:</p>
<p></p>
<p>````</p>
<p>{</p>
<p>"account_id": string,</p>
<p>"name": {</p>
<p>"given_name": string,</p>
<p>"surname": string,</p>
<p>"familiar_name": string,</p>
<p>"display_name": string,</p>
<p>"abbreviated_name": string</p>
<p>},</p>
<p>"email": string,</p>
<p>"email_verified": boolean,</p>
<p>"disabled": boolean,</p>
<p>"is_teammate": boolean,</p>
<p>"profile_photo_url": string</p>
<p>}</p>
<p>````</p>
<p></p>
<p>If an error occurred the JSON string will be in the following form:</p>
<p></p>
<p>````</p>
<p>{</p>
<p>"error_summary": string,</p>
<p>"error": {</p>
<p>".tag": string</p>
<p>}</p>
<p>}</p>
<p>````</p>
<p>*/</p>
<p></p>
<p>command dropboxGetAccount pAccessToken, pUserAccountID, pCallback</p>
<p>   local tHeader, tPost, tURL</p>
<p>   put __get_account_HEADER(pAccessToken) into tHeader</p>
<p>   put __get_account_POST(pUserAccountID) into tPost</p>
<p>   put __get_account_URL() into tUrl</p>
<p>   </p>
<p>   __API tHeader, tPost, tURL, pCallback</p>
<p>   if the result is empty then</p>
<p>      return it for value</p>
<p>   else</p>
<p>      return the result for error</p>
<p>   end if</p>
<p>end dropboxGetAccount</p>

<p>local sAsyncRequestA</p>
<p></p>
<p>private command __API pHeader, pPost, pURL, pCallback</p>
<p>   set socketTimeoutInterval to 30000  --30 seconds</p>
<p>   </p>
<p>   local tError</p>
<p>   if pCallback is not empty then</p>
<p>      local tRequestID</p>
<p>      put uuid() into tRequestID</p>
<p>      put __GetCaller() into sAsyncRequestA[tRequestID]["caller"]</p>
<p>      put pCallback into sAsyncRequestA[tRequestID]["callback"]</p>
<p>      put tsNetPost(tRequestID, pURL, pHeader, pPost, "__dropboxAPIAsyncCallback") into tError</p>
<p>      </p>
<p>      if tError is empty then</p>
<p>         return tRequestID for value</p>
<p>      end if</p>
<p>   else</p>
<p>      local tOldHeaders, tPostReturn</p>
<p>      put the httpHeaders into tOldHeaders</p>
<p>      set the httpHeaders to pHeader</p>
<p>      post pPost to URL pURL</p>
<p>      put it into tPostReturn</p>
<p>      put the result into tError</p>
<p>      set the httpHeaders to tOldHeaders</p>
<p>      </p>
<p>      if tError is empty then</p>
<p>         return tPostReturn for value</p>
<p>      end if</p>
<p>   end if</p>
<p>   </p>
<p>   return tError for error</p>
<p>end __API</p>
<p></p>
<p>on __dropboxAPIAsyncCallback pRequestID, pStatusCode, pBytes, pCurlResultCode</p>
<p>   if exists(sAsyncRequestA[pRequestID]["caller"]) then</p>
<p>      local tData, tError</p>
<p>      put tsNetRetrData(pRequestID, tError) into tData</p>
<p>      </p>
<p>      dispatch sAsyncRequestA[pRequestID]["callback"] to sAsyncRequestA[pRequestID]["caller"] with pRequestID, pStatusCode, tData, tError</p>
<p>   end if</p>
<p>   </p>
<p>   tsNetCloseConn pRequestID</p>
<p>end __dropboxAPIAsyncCallback</p>
<p>   </p>
<p>private function __GetCaller</p>
<p>   get item 1 to -3 of line -4 of the executionContexts</p>
<p>   if there is not an it then</p>
<p>      delete item -1 of it</p>
<p>   end if</p>
<p>   return it</p>
<p>end __GetCaller</p>

<p>private function __get_account_batch_POST pUserAccountIDs</p>
<p>   //pUserAccountIDs on separate lines</p>
<p>   /*</p>
<p>   GetAccountBatchArg</p>
<p>   account_ids List of (String) List of user account identifiers. Should not contain any duplicate account IDs.</p>
<p>   {"account_ids":["dbid:________________","dbid:_____________________"]}</p>
<p>   */</p>
<p>   if the number of lines of pUserAccountIDs =  1 then</p>
<p>      return __CB(__Q("account_ids") &amp; ":" &amp;  __SB(__Q(pUserAccountIDs)))</p>
<p>   else</p>
<p>      local tgabString</p>
<p>      put "{" &amp; __Q("account_ids") &amp; ":" &amp;  "["  into tgabString</p>
<p>      repeat for each line tAccountID in pUserAccountIDs</p>
<p>         put __Q(tAccountID) &amp; comma after tgabString</p>
<p>      end repeat</p>
<p>      delete last char of tgabString</p>
<p>      put "]"  after tgabString</p>
<p>      return tgabString</p>
<p>   end if</p>
<p>end __get_account_batch_POST</p>

<p>private function __FormatAddFolderMembers pMembers</p>
<p>   -- pMembers is an array  pMembers[1][email] or [1][dropbox_id], [1][accesslevel]</p>
<p>   /*"members": </p>
<p>   [</p>
<p>   { "member": {".tag": "email","email": "justin@example.com"},"access_level": {".tag": "editor"}},</p>
<p>   {"member": {".tag": "dropbox_id","dropbox_id": "dbid:AAEufNrMPSPe0dMQijRP0N_aZtBJRm26W4Q"},"access_level": {".tag": "editor"}}</p>

<p>   </p>

<p>      else</p>

<p>            put __Q("member") &amp; ":" after tString</p>
<p>            put __CB(__Q(".tag") &amp; ":" &amp; __Q("dropbox_id") &amp; "," &amp; __Q("dropbox_id") &amp; ":" &amp; __Q(pMembers[rKey]["dropbox_id"])) after tString</p>
<p>            put "," &amp; __Q("access_level") &amp; ":" &amp; __CB(__Q(".tag") &amp; ":" &amp; __Q(pMembers[rKey]["accesslevel"])) after tString</p>
<p>         end if</p>
<p>      end if</p>
<p>   end repeat</p>
<p>   put "]"  after tString</p>
<p>   return tString</p>
<p>   --</p>
<p>   return pMembers</p>

<p>private function __modify_shared_link_settings_POST pUrl,pSettings, pRemoveExpiration</p>
<p>   /*</p>
<p>   ModifySharedLinkSettingsArgs</p>
<p>   #--url String URL of the shared link to change its settings</p>
<p>   #--settings SharedLinkSettings Set of settings for the shared link.</p>
<p>   SharedLinkSettings</p>
<p>   ##----requested_visibility RequestedVisibility? The requested access for this shared link. This field is optional.</p>
<p>   RequestedVisibility (union)</p>
<p>   The access permission that can be requested by the caller for the shared link. Note that the final resolved visibility of the shared link takes into account other aspects, such as team and shared folder settings. Check the ResolvedVisibility for more info on the possible resolved visibility values of shared links. The value will be one of the following datatypes:</p>
<p>   ###------public Void Anyone who has received the link can access it. No login required.</p>
<p>   ###------team_only Void Only members of the same team can access the link. Login is required.</p>
<p>   ###------password Void A link-specific password is required to access the link. Login is not required.</p>
<p>   ##----link_password String? If requested_visibility is RequestedVisibility.password this is needed to specify the password to access the link. This field is optional.</p>
<p>   ##----expires Timestamp? Expiration time of the shared link. By default the link won't expire. This field is optional.</p>
<p>   */</p>
<p>   local tPost</p>
<p>   put __Q("url") &amp; ":" &amp;  __Q(pUrl) into tPost</p>
<p>   local tSettings</p>
<p>   repeat for each word tKey in "requested_visibility link_password expires"</p>
<p>      if pSettings[tKey] is not empty then</p>
<p>         if tSettings is not empty then</p>
<p>            put comma after tSettings</p>
<p>         end if</p>
<p>         put __Q(tKey) &amp; ":" &amp; pSettings[tKey] after tSettings</p>
<p>      end if</p>
<p>   end repeat</p>
<p>   </p>
<p>   if tSettings is not empty then</p>
<p>      put comma &amp; __Q("settings") &amp; ":" &amp; __CB(tSettings) after tPost</p>
<p>   end if</p>
<p>   </p>
<p>   if pRemoveExpiration is not empty then</p>
<p>      put comma &amp; __Q("remove_expiration") &amp; ":" &amp; pRemoveExpiration after tPost</p>
<p>   end if</p>
<p>   </p>
<p>   return __CB(tPost)</p>
<p>end __modify_shared_link_settings_POST</p>

<p>private function __remove_folder_member_POST pSharedFolderId,pMember,pLeaveACopy</p>
<p>   /*</p>
<p>   RemoveFolderMemberArg</p>
<p>   #--shared_folder_id String The ID for the shared folder.</p>
<p>   #--member MemberSelector The member to remove from the folder. Only the MemberSelector.dropbox_id may be set at this time.</p>
<p>   MemberSelector (open union)</p>
<p>   Includes different ways to identify a member to add to a shared folder.The value will be one of the following datatypes. </p>
<p>   New values may be introduced as our API evolves.</p>
<p>   ##----dropbox_id String Dropbox account, team member, or group ID of member to add.</p>
<p>   ##----email String E-mail address of member to add.</p>
<p>   #--leave_a_copy Boolean If true, the removed user will keep their copy of the folder after it's unshared, assuming it was mounted. </p>
<p>   Otherwise, it will be removed from their Dropbox. Also, this must be set to false when kicking a group.</p>
<p>   */</p>
<p>   </p>
<p>   local tMember</p>
<p>   if pMember["dropbox_id"] is not empty then</p>
<p>      put __Q(".tag") &amp; ":" &amp; __Q("dropbox_id"),__Q("dropbox_id") &amp; ":" &amp; __Q(pMember["dropbox_id"]) into tMember</p>
<p>   else</p>
<p>      put __Q(".tag") &amp; ":" &amp; __Q("email"),__Q("email") &amp; ":" &amp; __Q(pMember["email"]) into tMember</p>
<p>   end if</p>
<p>   return __CB(__Q("shared_folder_id") &amp; ":" &amp;  __Q(pSharedFolderId) &amp; "," &amp;__Q("member") &amp; ":" &amp; __CB(tMember) &amp; "," &amp;__Q("leave_a_copy") &amp; ":" &amp; pLeaveACopy)</p>
<p>end __remove_folder_member_POST</p>

<p>private function __share_folder_POST pPath,pMemberPolicy,pAclUpdatePolicy,pSharedLinkPolicy,pForceAsync</p>
<p>   /*</p>
<p>   ShareFolderArg</p>
<p>   #--path String The path to the folder to share. If it does not exist, then a new one is created.</p>
<p>   #--member_policy MemberPolicy Who can be a member of this shared folder.</p>
<p>   MemberPolicy (open union)</p>
<p>   Policy governing who can be a member of a shared folder. Only applicable to folders owned by a user on a team.The value will be one of the following datatypes. </p>
<p>   New values may be introduced as our API evolves.</p>
<p>   ##----team Void Only a teammate can become a member.</p>
<p>   ##----anyone Void Anyone can become a member.</p>
<p>   #--acl_update_policy AclUpdatePolicy Who can add and remove members of this shared folder.</p>
<p>   AclUpdatePolicy (open union)</p>
<p>   Policy governing who can change a shared folder's access control list (ACL). In other words, who can add, remove, or change the privileges of members.</p>
<p>   The value will be one of the following datatypes. New values may be introduced as our API evolves.</p>
<p>   ##----owner Void Only the owner can update the ACL.</p>
<p>   ##----editors Void Any editor can update the ACL. This may be further restricted to editors on the same team.</p>
<p>   #--shared_link_policy SharedLinkPolicy The policy to apply to shared links created for content inside this shared folder.</p>
<p>   SharedLinkPolicy (open union)</p>
<p>   Policy governing who can view shared links.The value will be one of the following datatypes. New values may be introduced as our API evolves.</p>
<p>   ##----anyone Void Links can be shared with anyone.</p>
<p>   ##----members Void Links can only be shared among members of the shared folder.</p>
<p>   #--force_async Boolean Whether to force the share to happen asynchronously. The default for this field is False.</p>
<p>   */</p>
<p>   local tPost</p>
<p>   put __Q("path") &amp; ":" &amp;  __Q(pPath) into tPost</p>
<p>   if pMemberPolicy is not empty then</p>
<p>      put "," &amp;__Q("member_policy") &amp; ":" &amp; __Q(pMemberPolicy) after tPost</p>
<p>   end if</p>
<p>   if pAclUpdatePolicy is not empty then</p>
<p>      put "," &amp;__Q("acl_update_policy") &amp; ":" &amp; __Q(pAclUpdatePolicy) after tPost</p>
<p>   end if</p>
<p>   if pSharedLinkPolicy is not empty then</p>
<p>      put "," &amp;__Q("shared_link_policy") &amp; ":" &amp; __Q(pSharedLinkPolicy) after tPost</p>
<p>   end if</p>
<p>   if pForceAsync is not empty then</p>
<p>      --default false</p>
<p>      put "," &amp;__Q("force_async") &amp; ":" &amp; __Q(pForceAsync) after tPost</p>
<p>   end if</p>
<p>   return __CB(tPost)</p>
<p>end __share_folder_POST</p>

<p>private function __FormatUpdateFolderMember pMember</p>
<p>   -- pMember is an array  pMember["email"] or pMember["dropbox_id"]</p>
<p>   /*</p>
<p>   --{".tag":"email","email":"myemailaddress"}</p>
<p>   --{".tag": "dropbox_id","dropbox_id": "dbid:mydropboxid"}</p>
<p>   */</p>
<p>   if pMember["email"] is not empty then </p>
<p>      return __CB(__Q(".tag") &amp; ":" &amp; __Q("email") &amp; "," &amp; __Q("email") &amp; ":" &amp; __Q(pMember["email"]))</p>
<p>   end if</p>
<p>   if pMember["dropbox_id"] is not empty then </p>
<p>      return __CB(__Q(".tag") &amp; ":" &amp; __Q("dropbox_id") &amp; "," &amp; __Q("dropbox_id") &amp; ":" &amp; __Q(pMember["dropbox_id"]))</p>
<p>   end if</p>
<p>   return pMember  --this is useful for syntax display and also allows preformatted json to be passed eg. {".tag":"email","email":"myemailaddress"}</p>

<p>private function __QInStr pString,pAltChar</p>
<p>   if pAltChar = empty then put "^" into pAltChar</p>
<p>   replace pAltChar with quote in pString</p>
<p>   return pString</p>
<p>end __QInStr</p>
<p></p>
<p>private function __Q pString</p>
<p>   return quote &amp; pString &amp; quote</p>
<p>end __Q</p>
<p></p>
<p>private function __CB pString</p>
<p>   return "{" &amp; pString &amp; "}"</p>
<p>end __CB</p>
<p></p>
<p>private function __SB pString</p>
<p>   return "[" &amp; pString &amp; "]"</p>
<p>end __SB</p>
<p></p>
<p>private function __List pString</p>
<p>   local tList</p>
<p>   repeat for each line tLine in pString</p>
<p>      if tList is not empty then </p>
<p>         put comma after tList</p>
<p>      end if</p>
<p>      put __Q(tLine) after tList</p>
<p>   end repeat</p>
<p>   return __SB(tList)</p>
<p>end __List</p>
